\section{Question 1}
\subsection{Part a}

\begin{equation*}
\begin{split}
    P_{atm} & = P_{m}gh  \\
        & = 13590  \cdot  3.7  \cdot  0.0119 \\
        & = 598.3677 \\
        & = 598.4 \text{Pa} \\
        & = \SI{5.984}{\milli\bar}
\end{split}
\end{equation*}

\subsection{Part b}

\begin{equation*}
    \begin{split}
        P_{hab} & = P_{atm} + \rho gh \\
                & = 598.3677 + 13590  \cdot  3.7  \cdot  2 \\
                & = 101.16 \text{x} 10^{3} \\
                & = 101.16 \text{kPa}
    \end{split}
\end{equation*}
