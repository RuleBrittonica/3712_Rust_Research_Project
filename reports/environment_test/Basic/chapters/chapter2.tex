    \begin{split}
        A & = 50  \cdot  45 + 2  \cdot  20  \cdot  45 + 2  \cdot  20  \cdot  45 \\
          & = \SI{6050}{\milli\meter\squared} \\
        E & = Q_{c} + Q_{e} \\

          & = [4.5 \cdot (33 - 22) + 0.92 \cdot 5.67\text{x}10^{-8} \cdot (306^4 - 293^4)] \cdot 6050\text{x}10^{-6} \\
